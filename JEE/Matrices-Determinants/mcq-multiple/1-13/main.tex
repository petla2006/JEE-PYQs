                \mydet{
                    a & b & a \alpha + b\\
                    b & c & b \alpha + c\\
                    a \alpha + b & b \alpha + c & 0

            \begin{enumerate}
                        \item $a,b,c$ are in A.P.
                        \item $a,b,c$ are in G.P.
                        \item $a,b,c$ are in H.P.
                        \item $\alpha$ is a root of the equation $ax^2 + bx +c=0$
